GRUNT FTP
=========

$ npm install --save-dev grunt-ftp

require('load-grunt-tasks')(grunt); // npm install --save-dev load-grunt-tasks 

upload :
--------

grunt.initConfig({
    ftpPut: {
        options: {
            host: 'website.com',
            user: 'johndoe',
            pass: '1234'
        },
        upload: {
            files: {
                'public_html': 'src/*'
            }
        }
    }
});
 
grunt.registerTask('default', ['ftpPut']);

download:
--------

require('load-grunt-tasks')(grunt); // npm install --save-dev load-grunt-tasks 
 
grunt.initConfig({
    ftpGet: {
        options: {
            host: 'website.com',
            user: 'johndoe',
            pass: '1234'
        },
        download: {
            files: {
                'public_html/file.txt': 'src/file.txt'
            }
        }
    }
});
 
grunt.registerTask('default', ['ftpGet']);

--

host
Required
Type: string

port
Type: number
Default: 21

user
Type: string
Default: 'anonymous'

pass
Type: string
Default: '@anonymous'
